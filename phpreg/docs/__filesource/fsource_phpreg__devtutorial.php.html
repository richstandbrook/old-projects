30    // B) Adding keys using the editor.
31    // 1. Browse to the phpregedit.php file on your server.
32    // 2. Type in the path to the registry file.
33    // 2.i. You can create a new file by entering a filename that doesnot exist
34    // !!!NOTE THAT PHPREG CANNOT CREATE SYSTEM DIRECTORIES!!!
35    // 3. at the top of the editor there are two text boxes, one button and a checkbox
36    //    i.   The first text box is the KEY, how you will identify the registry key.
37    //         This could be one word `GLOBALS` or a deep key `GLOBAL/DB/USERNAME` this structure
38    //         will be set up automaticly.
39    //    ii.  The second text box is the value, what they key is this can be any characters.
40    //    iii. The button will post the form to set the key
41    //    iv.  The checkbox can be ticked before submiting, to make the KEY an extension of the CURRENT path
42    //         if you are in the path `GLOABLS` and you set a key `DB` with the `extend key` box ticked this will
43    //         cause the structure `GLOBALS/DB` to be setup,
44    //         you can the browse to `GLOBALS/DB` and set keys like, `host` etc

60    // USING PHPREG IN YOUR APPLICATIONS
61    **************************************/
62    
63    //
64    // A) Opening a phpreg file
65    //
66    // 1. add the following code to your application, replace reg_file with the path to and name of your file
67    // if the file does not exsit, phpreg will atempt to create it.
68    // !!!NOTE THAT PHPREG CANNOT CREATE SYSTEM DIRECTORIES!!!
69    
70    <?php
71    
72    include('phpreg.php');
73    $reg = new phpreg;
74    
75    $reg->open("reg_file");
76    
77    ?>
78    
79    //
80    // B) Defining existing keys
81    //
82    // 1. follow the steps in A to open your file
83    // 2. Add the following code
84    
85    $reg->define_keys();
86    
87    // now any keys you have previously set will be defined as constants
88    // for example, if you set up a structure such as;
89    //
90    //  |- GLOBAL/
91    //  ¦        |- DB/
92    //  ¦          |- USER = root
93    //  ¦          |- HOST = localhost
94    //  ¦
95    //  |- VERSION = 0.1.2
96    //  ¦
97    //
98    // To access the key USER for the database you would do the following.
99    
100   echo constant("GLOBALS/DB/USER");
101   
102   // This would out put "root"
103   // You cannot call extended keys like this;
104   
105   echo GLOBALS/DB/USER;
106   
107   // The foward slashes `/` will trip up PHP and cause an error
108   // However, if you were to use a root level key you can call it without the constant() function
109   // using the structure above the following is posible
110   
111   echo VERSION
112   
113   // this would output "0.1.2"
114   
115   //
116   // C) Setting a new, or changing an exsiting, key
117   //
118   // 1. Follow the steps to open a file in part A
119   // 2. Use the following code to add a new key
120   
121   $reg->set_key("KEY_ID","key value");
122   
123   // you can also use the following
124   
125   $reg->set_key("EXTENDED/KEY_ID","key value");
126   
127   // the structure is automatically set up
128   // Leaving the VALUE blank
129   
130   $reg->set_key("KEY_ID");
131   
132   // This will create a extended key, this can also be done in the follwing way
133   
134   $reg->set_key("EXTENDED/KEY_ID");
135   
136   // TO BE ABLE TO CALL KEYS SET WITH set_key() YOU MUST THEN CALL define_keys()
137   
138   //
139   // D) Delete a key
140   //
141   // 1. Open a file (see A)
142   // 2. The following will work with keys from the file and set with set_key()
143   // !!! NOTE THAT ANY AND ALL CHILD KEY WILL BE REMOVED TO !!!
144   
145   $reg->delete_key("KEY_ID");
146   
147   // also use the extened format
148   
149   $reg->delete_key("EXTEND/KEY_ID");
150   
151   //
152   // E) Saving the phpreg file
153   //
154   // Up till now all the actions in "USING PHPREG IN YOUR APPLICATIONS"
155   // have been on the `running reg`, that is nothing has been saved to a file.
156   // when all changes have been made (or, indeed, when ever you want)
157   // you can use
158   
159   $reg->save("reg_file");
160   
161   // replacing `reg_file` with the path and name of the file to save to.
162   // if the file doesnt exsist phpreg will attempt to create it
163   // !!!NOTE THAT PHPREG CANNOT CREATE SYSTEM DIRECTORIES!!!
164
